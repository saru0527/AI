function onSuggestResolutionClick() {

    var sysId = g_form.getUniqueValue();
// sys_id of current record

    var shortDesc = g_form.getValue("short_description");



    g_form.addInfoMessage("üì§ Requesting resolution for ticket: " + sysId);



    var ga =
new GlideAjax('SuggestResolutionHelperAjax');

    ga.addParam('sysparm_name',
'getSuggestions');

    ga.addParam('sysparm_sys_id', sysId);

    ga.addParam('sysparm_query', shortDesc);



    ga.getXMLAnswer(function(response) {

        if (response) {

            g_form.addInfoMessage("‚úÖ AI Suggestion: " + response);

        } else {

            g_form.addErrorMessage("‚ùå No resolution suggestion received.");

        }

    });

}
